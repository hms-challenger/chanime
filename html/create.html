CREATE TABLE anime (
    AID INT PRIMARY KEY,
    titel TEXT,
    erschjahr TEXT,
    manga BOOLEAN, 
    SID INT,
    PID INT,
    AUTOR_ID INT,
    FOREIGN KEY (SID) REFERENCES status(SID),
    FOREIGN KEY (PID) REFERENCES produktionsfirma(PID),
    FOREIGN KEY (AUTOR_ID) REFERENCES autor(AUTOR_ID)
);
CREATE TABLE anime_genre (
    AID INT,
    GID INT,
    PRIMARY KEY (AID, GID),
    FOREIGN KEY (AID) REFERENCES anime(AID),
    FOREIGN KEY (GID) REFERENCES genre(GID)
);
CREATE TABLE autor (
    AUTOR_ID INT PRIMARY KEY,
    name text,
    geb date,
    tod date
);
    
CREATE TABLE beziehungen (
    BID int PRIMARY key,
    typ text
);
    
CREATE TABLE char_bez (
    CID1 INT,
    CID2 INT,
    BID INT,
    PRIMARY KEY (CID1, CID2, BID),
    FOREIGN KEY (CID1) REFERENCES charaktere(CID),
    FOREIGN KEY (CID2) REFERENCES charaktere(CID),
    FOREIGN KEY (BID) REFERENCES beziehungen(BID)
);
    
CREATE TABLE charaktere (
    CID INT PRIMARY KEY,
    name text,
    voice_actor text,
    signatur text,
    RID int, GID INT, 
    FOREIGN key (RID) REFERENCES rolle(RID)
);
    
CREATE TABLE genre (
    GID INT PRIMARY KEY,
    name text,
    beschreibung text
);
    
CREATE TABLE gruppe (
    GID int PRIMARY KEY,
    name text,
    leader text
);
    
CREATE TABLE produktionsfirma (
    PID INT PRIMARY KEY,
    name text,
    gruendungsjahr text,
    jahresumsatz int
);
    
CREATE TABLE rolle (
    RID INT PRIMARY KEY,
    typ text
);
    
CREATE TABLE status (
    SID INT PRIMARY KEY,
    typ text
